JavaScript funkce s argumentem

Co je argument a parametr v JavaScriptu? Argument je ta samotná hodnota, se kterou funkci voláme. Parametr je původní proměnná, se kterou se funkce definována.
Jak vytvořit funkci v JavaScriptu

var pocet = 0;
function spocitejsamohlasky(jmeno)
{
for (var i=0; i<jmeno.length; i++)
{
if(jmeno[i] == "a" || jmeno[i] == "e" || jmeno[i] == "i" || jmeno[i] == "o" || jmeno[i] == "u")
pocet = pocet + 1;
}
document.write("Jméno " + jmeno + " má " + pocet + " samohlásek");
}
var mojejmeno = prompt("Zadej své jméno : ");
spocitejsamohlasky(mojejmeno);
