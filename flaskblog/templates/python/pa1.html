OS modul - základní funkčnost operačního systému


Tímto kódem vytisknete aktulání složku ve které se nacházíte
import os
print(os.getcwd())

Tímto kódem změníte vytisknete aktulání složku ve které se nacházíte
os.chdir("/Users/Alesk/Desktop")
print(os.getcwd())

Tímto kódem vytisknete všechny složky a soubory v uvedené složce
os.chdir("/Users/Alesk/Desktop")
print(os.listdir())

Tímto kódem vytvoříme novou složku, nebo vice slozek
os.chdir("/Users/Alesk/Desktop")
os.mkdir("prvni slozka")
os.makedirs('druha slozka/slozka ve druhe slozce')

Tímto kódem složku(složky) smažeme
os.chdir("/Users/Alesk/Desktop")
os.rmdir("prvni slozka")
os.removedirs('druha slozka/slozka ve druhe slozce')

Tímto kódem přejmenujeme soubor test.txt na test1.txt
os.chdir("/Users/Alesk/Desktop")
os.chdir("/Users/Alesk/Desktop")
os.rename("test.txt","test1.txt")

Tímto kódem vytiskneme prodrobnosti souboru test.txt
import os
from datetime import datetime

os.chdir("/Users/Alesk/Desktop")
print(os.stat('test.txt'))
x = os.stat('cs1.txt').st_size
print(x)
y = os.stat('test.txt').st_mtime
print(datetime.fromtimestamp(y))

Tímto kódem vytiskneme pomocí cyklu "for" všechny soubory ve složkách a sub složkách
for dir_cesta,dir_nazev,nazev_souboru in os.walk("/Users/Alesk/Desktop"):
   print(dir_cesta)
   print(dir_nazev)
   print(nazev_souboru)

Tímto kódem pomocí "os.path.join" vytoříme nový soubor ve složce
cesta_souboru = os.path.join("/Users/Alesk/Desktop",'test.txt')
print(cesta_souboru)
